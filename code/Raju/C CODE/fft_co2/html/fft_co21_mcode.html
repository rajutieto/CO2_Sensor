1 function snrdB = fft_co2(X)
 2% fft_co2: - returns the noise amount in Decibel contained in time domain input signal.
 3
 4% Syntax: snrdB= fft_co2(X)
 5%
 6% Inputs:
 7%    X          - Time domain input signal
 8%
 9% Outputs:
10%    snrdB      - Noise amount in Decibel
11%
12
13%------------- BEGIN CODE --------------
14
15
16N=64;
17[m n]= size(X);
18
19% If the length of X is greater than n, the sequence X is truncated.
20if m>N
21    Xmod= X(1:N,:);
22% If the length of X is less than N, X is padded with trailing zeros to
23% length N
24elseif m<N
25    Xmod =[X ;zeros(N-m,n)];
26    
27elseif m==N
28    Xmod=X;
29end
30
31% j   = sqrt(-1);
32j   = mpower(-1,0.5);
33tk  = 0:(N-1);
34fn  = transpose(0:(N-1))/N;
35wn  = 2*pi*fn;
36
37p= -j*wn*tk;
38
39Wp  = exp(p); %DFT Matrix
40Y   = Wp*Xmod;
41
42delta=160e-3;
43fs=1/delta;
44FFT_sampled_data=1/m*Y;
45absFFT_sampled_data=abs(FFT_sampled_data);
46signal_value=max(absFFT_sampled_data);
47noise_mean=(sum(absFFT_sampled_data)-signal_value)/(m-1);    
48snrVal=signal_value/noise_mean;
49snrdB=10*log10(snrVal); 
50
51 end
52
53%------------- END OF CODE --------------
